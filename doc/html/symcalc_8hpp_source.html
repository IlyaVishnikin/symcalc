291  std::vector<std::string> string_to_rpn(std::string formula)
292  {
293  using namespace std;
294 
295  vector<string> output;
296  stack<string> stack;
297  size_t formula_length = formula.length();
298  for (size_t i = 0; i < formula_length; i++)
299  if (formula[i] == ' ' || formula[i] == ',')
300  continue;
301  else if (isdigit(formula[i]))
302  {
303  string number = formula.substr(i++, 1);
304 
305  // Number consists only of digits or '.'(decimal separator) characters
306  while (i < formula_length && (formula[i] == '.' || isdigit(formula[i])))
307  number += formula[i++];
308  output.push_back(number);
309  i--; // used in order not to skip next character
310  }
311  else if (formula[i] == '(')
312  stack.push("(");
313  else if (formula[i] == ')')
314  {
315  if (!stack.size())
316  throw length_error("Parenthesis at '" + to_string(i) + "' position was never been opened.");
317  while (stack.top() != "(")
318  {
319  output.push_back(stack.top());
320  stack.pop();
321  if (!stack.size())
322  throw length_error("Parenthesis at '" + to_string(i) + "' position was never been opened.");
323  }
324  stack.pop(); // remove left parenthesis from stack
325 
326  /* If operator writes not on infix form than parenthesis
327  * refer to the function(such as sqrt, cos, sin, etc.)
328  */
329  if (stack.size() && !Operators::is_infix(stack.top()))
330  {
331  output.push_back(stack.top());
332  stack.pop();
333  }
334  }
335  else // when it operator(function) or constant
336  {
337  string name = formula.substr(i++, 1);
338  bool is_punct = ispunct(name[0]);
339 
340  /* If symbol that we read is a punctuation character
341  * than operator or constant name consists only of punctuation characters.
342  * We should except '(' and ')' character because this is special characters.
343  *
344  * Otherwise, operator or constant name consist only of alphabet characters,
345  */
346  while (i < formula_length && (is_punct ? (ispunct(formula[i]) && formula[i] != '(' && formula[i] != ')') : isalpha(formula[i])))
347  name += formula[i++];
348  i--; // used in order not to skip next character
349 
350  if (Operators::is_operator(name))
351  {
352  if (i == formula_length)
353  throw invalid_argument("Operator \"" + name + "\" expects operands");
354 
355  if (!stack.size() || stack.top() == "(")
356  {
357  stack.push(name);
358  continue;
359  }
360 
361  /* Shunting yard algorithm.
362  * For more information see https://en.wikipedia.org/wiki/Shunting_yard_algorithm
363  * or main page in the documentation.
364  */
365  Operators::Precedence op1_precedence = Operators::get_precedence(name);
366  Operators::Precedence op2_precedence = Operators::get_precedence(stack.top());
367  bool is_op1_left_assocated = Operators::get_associativity(name) == Operators::LEFT_ASSOCIATED;
368  while (!stack.empty() &&
369  (op2_precedence > op1_precedence || (op1_precedence == op2_precedence && is_op1_left_assocated)))
370  {
371  output.push_back(stack.top());
372  stack.pop();
373  op2_precedence = Operators::get_precedence(stack.top());
374  }
375  stack.push(name);
376  }
377  else if (Consts::is_const(name))
378  output.push_back(to_string(Consts::get_const(name)));
379  else
380  throw invalid_argument("Symbol \"" + name + "\" is not operator or constant");
381  }
382 
383  /* Last step of shunting yard algorithm.
384  * For more information see https://en.wikipedia.org/wiki/Shunting_yard_algorithm
385  * or main page in the documentation.
386  */
387  while (stack.size())
388  {
389  output.push_back(stack.top());
390  stack.pop();
391  }
392 
393  return output;
394  }

402  std::vector<long double> calculate_rpn(std::vector<std::string> rpn)
403  {
404  using namespace std;
405 
406  vector<long double> output;
407  for (const string& s : rpn)
408  if (!Operators::is_operator(s))
409  output.push_back(stod(s));
410  else
411  {
412  /* Extracting @args_count passed arguments for operator(function).
413  *
414  * Note:
415  * Number of operator(function) arguments are varies.
416  * That means that in function can be passed 1, 2, ..., max_args_count arguments.
417  */
418  vector<long double> args;
419  Operators::ArgsCount args_count = Operators::get_args_count(s);
420  while (args_count-- && output.size())
421  {
422  args.push_back(output[output.size()-1]);
423  output.pop_back();
424  }
425 
426  /* Flipping extracted arguments vector, because order of passed arguments
427  * to operator(function) in RPN notation has been reversed.
428  */
429  size_t args_length = args.size();
430  for (size_t i = 0; i < args_length/2; i++)
431  {
432  long double temp = args[i];
433  args[i] = args[args_length-i-1];
434  args[args_length-i-1] = temp;
435  }
436 
437  vector<long double> operator_output = Operators::get_operator(s)(args);
438  while (operator_output.size())
439  {
440  output.push_back(operator_output[operator_output.size()-1]);
441  operator_output.pop_back();
442  }
443  }
444 
445  /* Reversing output vector */
446  size_t output_length = output.size();
447  for (size_t i = 0; i < output.size() / 2; i++)
448  {
449  long double temp = output[i];
450  output[i] = output[output_length-i];
451  output[output_length-i] = temp;
452  }
453  return output;
454  }
